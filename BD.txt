-- ---
-- Table 'USER'
--

DROP TABLE IF EXISTS `USER`;
		
CREATE TABLE `USER` (
  `ID_USER` int NOT NULL AUTO_INCREMENT DEFAULT NULL,
  `NAME` varchar NOT NULL DEFAULT 'NULL',
  `EMAIL` varchar NOT NULL DEFAULT 'NULL',
  `PASSWORD` tinyint NOT NULL DEFAULT NULL,
  PRIMARY KEY (`ID_USER`)
);

-- ---
-- Table 'ADDRESS'
-- 
-- ---

DROP TABLE IF EXISTS `ADDRESS`;
		
CREATE TABLE `ADDRESS` (
  `ID_ADDRESS` int NOT NULL AUTO_INCREMENT DEFAULT NULL,
  `STREET` varchar NOT NULL DEFAULT 'NULL',
  `NUMBER` int NOT NULL DEFAULT NULL,
  `COMPLEMENT` varchar NULL DEFAULT NULL,
  `ZIP_CODE` int NULL DEFAULT NULL,
  `NEIGHBORHOOD` varchar NOT NULL DEFAULT 'NULL',
  `CITY` varchar NULL DEFAULT NULL,
  `STATE` varchar NOT NULL DEFAULT 'NULL',
  PRIMARY KEY (`ID_ADDRESS`)
);

-- ---
-- Table 'TYPE_OF_USER'
-- 
-- ---

DROP TABLE IF EXISTS `TYPE_OF_USER`;
		
CREATE TABLE `TYPE_OF_USER` (
  `ID_TYPE_OF_USER` int NULL AUTO_INCREMENT DEFAULT NULL,
  `TYPE` sql_variant NULL DEFAULT NULL COMMENT 'ADMINISTRATOR, CANDIDATE',
  PRIMARY KEY (`ID_TYPE_OF_USER`)
);

-- ---
-- Table 'CANDIDATE DATA'
-- 
-- ---

DROP TABLE IF EXISTS `CANDIDATE DATA`;
		
CREATE TABLE `CANDIDATE DATA` (
  `ID_CANDIDATE DATA` int NULL AUTO_INCREMENT DEFAULT NULL,
  `DATE_OF_BIRTH` datetime NOT NULL DEFAULT NULL,
  `CPF` varchar NULL DEFAULT NULL,
  `BEEN_CONFIRMED` varchar NULL DEFAULT NULL,
  PRIMARY KEY (`ID_CANDIDATE DATA`)
);
